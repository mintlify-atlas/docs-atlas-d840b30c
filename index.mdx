    <h2 className="text-3xl font-bold text-gray-900 dark:text-white mb-4">Quick Start</h2>
    <p className="text-lg text-gray-600 dark:text-gray-400">Get up and running with UI Kitten in minutes</p>
  </div>

  <Steps>
    <Step title="Install UI Kitten">
      Create a new React Native project with UI Kitten template or add it to an existing project.

      <CodeGroup>
      ```bash npm
      npx react-native init MyApp --template @ui-kitten/template-ts
      ```

      ```bash yarn
      npx react-native init MyApp --template @ui-kitten/template-ts
      ```
      </CodeGroup>

      <Note>Use `@ui-kitten/template-js` for JavaScript projects</Note>
    </Step>

    <Step title="Wrap your app with ApplicationProvider">
      Import the ApplicationProvider and Eva themes to enable UI Kitten components.

      ```tsx
      import React from 'react';
      import { ApplicationProvider } from '@ui-kitten/components';
      import * as eva from '@eva-design/eva';

      export default () => (
        <ApplicationProvider {...eva} theme={eva.light}>
          {/* Your app content */}
        </ApplicationProvider>
      );
      ```
    </Step>

    <Step title="Use UI Kitten components">
      Start building your interface with ready-to-use components.

      ```tsx
      import React from 'react';
      import { Layout, Text, Button } from '@ui-kitten/components';

      export const HomeScreen = () => (
        <Layout style={{ flex: 1, justifyContent: 'center', alignItems: 'center' }}>
          <Text category='h1'>Welcome to UI Kitten</Text>
          <Button onPress={() => {}}>Get Started</Button>
        </Layout>
      );
      ```

      <Tip>All components support theming, styling variants, and status modifiers out of the box</Tip>
    </Step>
  </Steps>
